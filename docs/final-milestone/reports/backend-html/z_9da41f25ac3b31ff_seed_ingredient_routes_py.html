95class Command(BaseCommand): 

96 help = 'Seed IngredientRoute migration waypoints for 10 iconic ingredients (idempotent).' 

97 

98 def handle(self, *args, **options): 

99 created = 0 

100 updated = 0 

101 skipped_unchanged = 0 

102 

103 for name, waypoints in SEED_ROUTES.items(): 

104 ingredient, _ = Ingredient.objects.get_or_create( 

105 name=name, 

106 defaults={'is_approved': True}, 

107 ) 

108 route = IngredientRoute.objects.filter(ingredient=ingredient).first() 

109 if route is None: 

110 IngredientRoute.objects.create(ingredient=ingredient, waypoints=waypoints) 

111 created += 1 

112 continue 

113 

114 if route.waypoints == waypoints: 

115 skipped_unchanged += 1 

116 continue 

117 

118 route.waypoints = waypoints 

119 route.save(update_fields=['waypoints', 'updated_at']) 

120 updated += 1 

121 

122 self.stdout.write(self.style.SUCCESS( 

123 f'Ingredient routes seeded: {created} created, {updated} updated, {skipped_unchanged} unchanged.' 

124 ))
