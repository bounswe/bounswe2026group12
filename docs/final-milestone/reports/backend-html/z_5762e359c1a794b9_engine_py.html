1"""Pure unit conversion engine. 

2 

3No Django imports, no DB. Operates on Decimal to avoid float drift. 

4 

5All ratios resolve to a base unit per dimension: 

6- Mass: grams (g) 

7- Volume: millilitres (ml) 

8 

9Mass <-> volume crossing requires a density value in g/ml. 

10""" 

11from __future__ import annotations 

12 

13from decimal import Decimal 

14from typing import Mapping 

15 

16 

17class ConversionError(ValueError): 

18 """Raised for any unconvertible request: unknown unit, missing density, etc.""" 

19 

20 

21# Aliases normalize user input. Values are the canonical keys used in the ratio 

22# tables below. Lower-cased, whitespace-stripped, dot-stripped. 

23_ALIASES: Mapping[str, str] = { 

24 # mass: metric 

25 'mg': 'mg', 'milligram': 'mg', 'milligrams': 'mg', 

26 'g': 'g', 'gram': 'g', 'grams': 'g', 

27 'kg': 'kg', 'kilogram': 'kg', 'kilograms': 'kg', 

28 # mass: imperial 

29 'oz': 'oz', 'ounce': 'oz', 'ounces': 'oz', 

30 'lb': 'lb', 'lbs': 'lb', 'pound': 'lb', 'pounds': 'lb', 

31 # volume: metric 

32 'ml': 'ml', 'millilitre': 'ml', 'millilitres': 'ml', 'milliliter': 'ml', 'milliliters': 'ml', 

33 'l': 'l', 'litre': 'l', 'litres': 'l', 'liter': 'l', 'liters': 'l', 

34 # volume: imperial / US customary 

35 'tsp': 'tsp', 'teaspoon': 'tsp', 'teaspoons': 'tsp', 

36 'tbsp': 'tbsp', 'tablespoon': 'tbsp', 'tablespoons': 'tbsp', 

37 'fl_oz': 'fl_oz', 'floz': 'fl_oz', 'fl oz': 'fl_oz', 'fluid ounce': 'fl_oz', 'fluid ounces': 'fl_oz', 

38 'cup': 'cup', 'cups': 'cup', 

39 'pint': 'pint', 'pints': 'pint', 'pt': 'pint', 

40 'quart': 'quart', 'quarts': 'quart', 'qt': 'quart', 

41 'gallon': 'gallon', 'gallons': 'gallon', 'gal': 'gallon', 

42 # informal volume 

43 'pinch': 'pinch', 'pinches': 'pinch', 

44 'dash': 'dash', 'dashes': 'dash', 

45 'smidgen': 'smidgen', 'smidgens': 'smidgen', 

46} 

47 

48 

49# Mass to grams. 

50# Imperial: 1 oz = 28.349523125 g (international avoirdupois, NIST). 

51_MASS_TO_G: Mapping[str, Decimal] = { 

52 'mg': Decimal('0.001'), 

53 'g': Decimal('1'), 

54 'kg': Decimal('1000'), 

55 'oz': Decimal('28.349523125'), 

56 'lb': Decimal('453.59237'), 

57} 

58 

59 

60# Volume to millilitres. 

61# US customary: 1 cup = 240 ml (legal/nutrition labelling), 1 tbsp = 15 ml, 

62# 1 tsp = 5 ml. fl_oz, pint, quart, gallon use the US legal cup of 240 ml as 

63# the basis (8 fl_oz = 1 cup => 1 fl_oz = 30 ml; 2 cups = 1 pint, 4 cups = 

64# 1 quart, 16 cups = 1 gallon). This matches FDA labelling rather than the 

65# slightly-smaller US customary cup (236.588 ml). Picked deliberately because 

66# recipe contexts almost always assume the labelling cup. 

67# 

68# Informal: 

69# pinch = 1/16 tsp 

70# dash = 1/8 tsp 

71# smidgen = 1/32 tsp 

72# These are the widely-accepted measuring-spoon-set ratios. 

73_VOLUME_TO_ML: Mapping[str, Decimal] = { 

74 'ml': Decimal('1'), 

75 'l': Decimal('1000'), 

76 'tsp': Decimal('5'), 

77 'tbsp': Decimal('15'), 

78 'fl_oz': Decimal('30'), 

79 'cup': Decimal('240'), 

80 'pint': Decimal('480'), 

81 'quart': Decimal('960'), 

82 'gallon': Decimal('3840'), 

83 'pinch': Decimal('5') / Decimal('16'), 

84 'dash': Decimal('5') / Decimal('8'), 

85 'smidgen': Decimal('5') / Decimal('32'), 

86}

89def _normalize(unit: str) -> str: 

90 if not isinstance(unit, str): 

91 raise ConversionError(f'Unit must be a string, got {type(unit).__name__}.') 

92 key = unit.strip().lower().replace('.', '') 

93 canonical = _ALIASES.get(key) 

94 if canonical is None: 

95 raise ConversionError(f'Unknown unit: {unit!r}.') 

96 return canonical 

97 

98 

99def _dimension(canonical: str) -> str: 

100 if canonical in _MASS_TO_G: 

101 return 'mass' 

102 if canonical in _VOLUME_TO_ML: 

103 return 'volume' 

104 raise ConversionError(f'Unknown unit: {canonical!r}.') 

105 

106 

107def convert( 

108 amount, 

109 from_unit: str, 

110 to_unit: str, 

111 *, 

112 density_g_per_ml: Decimal | None = None, 

113) -> Decimal: 

114 """Convert ``amount`` from ``from_unit`` to ``to_unit``. 

115 

116 Mass <-> volume conversion requires ``density_g_per_ml``. 

117 Returns a Decimal. Raises ConversionError on any failure. 

118 """ 

119 try: 

120 amount_d = Decimal(str(amount)) 

121 except Exception as exc: 

122 raise ConversionError(f'Invalid amount: {amount!r}.') from exc 

123 

124 if amount_d < 0: 

125 raise ConversionError('Amount must be non-negative.') 

126 

127 src = _normalize(from_unit) 

128 dst = _normalize(to_unit) 

129 

130 src_dim = _dimension(src) 

131 dst_dim = _dimension(dst) 

132 

133 if src_dim == dst_dim: 

134 return _convert_same_dim(amount_d, src, dst, src_dim) 

135 

136 if density_g_per_ml is None: 

137 raise ConversionError( 

138 'Mass to volume conversion requires density_g_per_ml; ' 

139 f'cannot convert {from_unit!r} to {to_unit!r} without it.' 

140 ) 

141 try: 

142 density = Decimal(str(density_g_per_ml)) 

143 except Exception as exc: 

144 raise ConversionError(f'Invalid density: {density_g_per_ml!r}.') from exc 

145 if density <= 0: 

146 raise ConversionError('density_g_per_ml must be positive.') 

147 

148 if src_dim == 'volume' and dst_dim == 'mass': 

149 ml = amount_d * _VOLUME_TO_ML[src] 

150 grams = ml * density 

151 return grams / _MASS_TO_G[dst] 

152 

153 # mass -> volume 

154 grams = amount_d * _MASS_TO_G[src] 

155 ml = grams / density 

156 return ml / _VOLUME_TO_ML[dst] 

157 

158 

159def _convert_same_dim(amount: Decimal, src: str, dst: str, dim: str) -> Decimal: 

160 table = _MASS_TO_G if dim == 'mass' else _VOLUME_TO_ML 

161 base = amount * table[src] 

162 return base / table[dst]
