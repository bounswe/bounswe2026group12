30def _annotate_region_counts(qs): 

31 """Annotate a Region queryset with published content counts. 

32 

33 Counts are for published recipes, published stories, and active 

34 cultural content that reference each region. 

35 """ 

36 return qs.annotate( 

37 recipe_count=Count( 

38 'recipes', 

39 filter=Q(recipes__is_published=True), 

40 distinct=True, 

41 ), 

42 story_count=Count( 

43 'stories', 

44 filter=Q(stories__is_published=True), 

45 distinct=True, 

46 ), 

47 cultural_count=Count( 

48 'cultural_content', 

49 filter=Q(cultural_content__is_active=True), 

50 distinct=True, 

51 ), 

52 )

59class RegionIndexView(APIView): 

60 """GET /api/map/regions/ 

61 

62 Returns all regions that have center coordinates set, with content counts. 

63 This is the primary feed for placing pins/markers on the map. 

64 

65 Query params: 

66 geo_only (bool, default true): if false, return all regions even those 

67 without coordinates (useful for admin validation). 

68 """ 

69 permission_classes = [permissions.AllowAny] 

70 

71 def get(self, request): 

72 geo_only = request.query_params.get('geo_only', 'true').lower() != 'false' 

73 qs = Region.objects.select_related('parent').all() 

74 if geo_only: 

75 qs = qs.filter(latitude__isnull=False, longitude__isnull=False) 

76 qs = _annotate_region_counts(qs).order_by('name') 

77 serializer = RegionGeoWithCountsSerializer(qs, many=True) 

78 return Response(serializer.data)

98class RegionContentView(APIView): 

99 """GET /api/map/regions/<id>/content/ 

100 

101 Returns paginated content (recipes, stories, cultural content) tagged to 

102 a specific region. 

103 

104 Query params: 

105 type (str): filter by content type — 'recipe', 'story', or 'cultural'. 

106 Omit to receive all types interleaved by creation date. 

107 page (int): page number (default 1). 

108 page_size (int): items per page (default 20, max 100). 

109 """ 

110 permission_classes = [permissions.AllowAny] 

111 

112 def get(self, request, pk): 

113 try: 

114 region = Region.objects.get(pk=pk) 

115 except Region.DoesNotExist: 

116 return Response({'detail': 'Region not found.'}, status=status.HTTP_404_NOT_FOUND) 

117 

118 content_type = request.query_params.get('type', '').strip().lower() 

119 paginator = MapPageNumberPagination() 

120 

121 results = [] 

122 

123 if not content_type or content_type == 'recipe': 

124 recipes = ( 

125 Recipe.objects 

126 .filter(region=region, is_published=True) 

127 .select_related('region', 'author') 

128 .order_by('-created_at') 

129 ) 

130 # Apply limit before evaluating to avoid loading thousands of objects 

131 # page_size * current_page is the maximum index we would need 

132 page_number = int(request.query_params.get('page', 1)) 

133 max_limit = page_number * paginator.page_size 

134 results += list(MapRecipeCardSerializer(recipes[:max_limit], many=True).data) 

135 

136 if not content_type or content_type == 'story': 

137 # Stories tagged directly to the region OR inheriting via linked recipes 

138 stories = ( 

139 Story.objects 

140 .filter( 

141 Q(region=region) | Q(recipe_links__recipe__region=region), 

142 is_published=True, 

143 ) 

144 .select_related('author', 'region') 

145 .prefetch_related('recipe_links__recipe__region') 

146 .distinct() 

147 .order_by('-created_at') 

148 ) 

149 page_number = int(request.query_params.get('page', 1)) 

150 max_limit = page_number * paginator.page_size 

151 results += list(MapStoryCardSerializer(stories[:max_limit], many=True).data) 

152 

153 if not content_type or content_type == 'cultural': 

154 cultural = ( 

155 CulturalContent.objects 

156 .filter(region=region, is_active=True) 

157 .select_related('region') 

158 .order_by('-created_at') 

159 ) 

160 page_number = int(request.query_params.get('page', 1)) 

161 max_limit = page_number * paginator.page_size 

162 results += list(MapCulturalCardSerializer(cultural[:max_limit], many=True).data) 

163 

164 # Sort unified results by created_at descending (cultural content has no 

165 # created_at if type is cultural-only, handled gracefully) 

166 results.sort(key=lambda x: x.get('created_at', ''), reverse=True) 

167 

168 page = paginator.paginate_queryset(results, request) 

169 if page is not None: 

170 return paginator.get_paginated_response(page) 

171 return Response({ 

172 'region': RegionGeoSerializer(region).data, 

173 'results': results, 

174 'count': len(results), 

175 })

178class BoundingBoxDiscoverView(APIView): 

179 """GET /api/map/discover/ 

180 

181 Returns regions (with content counts) whose center point falls within the 

182 given viewport bounding box. This is the primary endpoint called when the 

183 user pans or zooms the map. 

184 

185 Required query params: 

186 north (float): northern latitude bound 

187 south (float): southern latitude bound 

188 east (float): eastern longitude bound 

189 west (float): western longitude bound 

190 

191 Optional query params: 

192 type (str): 'recipe', 'story', or 'cultural' — filters content counts 

193 to only the specified type (does not change which regions appear). 

194 q (str): text filter — only include regions that have at least one 

195 piece of content matching the keyword. 

196 """ 

197 permission_classes = [permissions.AllowAny] 

198 

199 PARAM_ERROR = ( 

200 'Provide north, south, east, west as numeric latitude/longitude bounds.' 

201 ) 

202 

203 def _parse_bbox(self, params): 

204 """Parse and validate bounding box from query params. 

205 

206 Returns (north, south, east, west) floats or raises ValueError. 

207 """ 

208 try: 

209 north = float(params['north']) 

210 south = float(params['south']) 

211 east = float(params['east']) 

212 west = float(params['west']) 

213 except (KeyError, TypeError, ValueError): 

214 raise ValueError(self.PARAM_ERROR) 

215 

216 if not (-90 <= south <= north <= 90): 

217 raise ValueError('south must be ≤ north, both in [-90, 90].') 

218 if not (-180 <= west <= 180) or not (-180 <= east <= 180): 

219 raise ValueError('east and west must be in [-180, 180].') 

220 

221 return north, south, east, west 

222 

223 def get(self, request): 

224 try: 

225 north, south, east, west = self._parse_bbox(request.query_params) 

226 except ValueError as exc: 

227 return Response({'detail': str(exc)}, status=status.HTTP_400_BAD_REQUEST) 

228 

229 content_type = request.query_params.get('type', '').strip().lower() 

230 keyword = request.query_params.get('q', '').strip() 

231 

232 # Find regions whose center is inside the viewport rectangle. 

233 # We only return regions that have coordinates (pins only make sense 

234 # when we know where to put them). 

235 qs = Region.objects.filter( 

236 latitude__isnull=False, 

237 longitude__isnull=False, 

238 latitude__lte=north, 

239 latitude__gte=south, 

240 ).select_related('parent') 

241 

242 # Longitudinal bounding box wrap-around logic (Pacific antimeridian) 

243 if east < west: 

244 # Viewport crosses the antimeridian (e.g., west=170, east=-170) 

245 qs = qs.filter(Q(longitude__gte=west) | Q(longitude__lte=east)) 

246 else: 

247 # Standard viewport 

248 qs = qs.filter(longitude__lte=east, longitude__gte=west) 

249 

250 # Optional keyword filter: only include regions with matching content 

251 if keyword: 

252 qs = qs.filter( 

253 Q(recipes__title__icontains=keyword, recipes__is_published=True) | 

254 Q(stories__title__icontains=keyword, stories__is_published=True) | 

255 Q(stories__recipe_links__recipe__title__icontains=keyword, stories__is_published=True) | 

256 Q(cultural_content__title__icontains=keyword, cultural_content__is_active=True) 

257 ).distinct() 

258 

259 qs = _annotate_region_counts(qs).order_by('name') 

260 

261 serializer = RegionGeoWithCountsSerializer(qs, many=True) 

262 total_content = sum( 

263 (r['content_count']['recipes'] + 

264 r['content_count']['stories'] + 

265 r['content_count']['cultural_content']) 

266 for r in serializer.data 

267 ) 

268 

269 return Response({ 

270 'viewport': { 

271 'north': north, 'south': south, 

272 'east': east, 'west': west, 

273 }, 

274 'regions': serializer.data, 

275 'total_content': total_content, 

276 })
