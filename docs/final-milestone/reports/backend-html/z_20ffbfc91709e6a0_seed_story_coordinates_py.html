1"""Idempotent seeder for Story latitude/longitude coordinates (#757). 

2 

3Closes the gap that makes story pins invisible on the region map: Story rows 

4have nullable latitude/longitude fields (added in #730), but no existing story 

5row has them populated — so every GET /api/stories/?region={id} returns 

6lat=None, lng=None and no pins render. 

7 

8Run this command *after* seed_canonical AND seed_region_geo: 

9 

10 python manage.py seed_canonical 

11 python manage.py seed_region_geo # populates Region bbox fields 

12 python manage.py seed_story_coordinates # this command 

13 

14For every Story where both latitude and longitude are null: 

15 - If the story's region has a bounding box, assign a deterministic point 

16 inside that bbox (RNG seeded with the story id, so reruns are stable). 

17 - If the story has no region, or the region has no bbox, leave it null. 

18 

19Stories that already have coordinates are skipped (idempotent). 

20""" 

21import random 

22from decimal import Decimal 

23 

24from django.core.management.base import BaseCommand 

25 

26from apps.stories.models import Story 

27 

28 

29def _point_in_bbox(region, rng): 

30 """Return a deterministic (lat, lng) Decimal pair inside the region bbox. 

31 

32 Quantized to 6 decimal places to match Story.latitude/longitude 

33 (DecimalField, max_digits=9, decimal_places=6). 

34 """ 

35 lat = rng.uniform(region.bbox_south, region.bbox_north) 

36 lng = rng.uniform(region.bbox_west, region.bbox_east) 

37 q = Decimal('0.000001') 

38 return Decimal(repr(lat)).quantize(q), Decimal(repr(lng)).quantize(q) 

39 

40 

41class Command(BaseCommand): 

42 help = ( 

43 'Seed Story latitude/longitude from the story region bounding box. ' 

44 'Idempotent — skips rows that already have coordinates. ' 

45 'Run after seed_canonical and seed_region_geo.' 

46 ) 

47 

48 def handle(self, *args, **options): 

49 seeded, already_set, no_region, no_bbox = self._seed_stories() 

50 self.stdout.write(self.style.SUCCESS( 

51 f'seed_story_coordinates: seeded {seeded} stories. ' 

52 f'Skipped: {already_set} already had coords, ' 

53 f'{no_region} had no region, ' 

54 f'{no_bbox} had no region bbox.' 

55 )) 

56 

57 def _seed_stories(self): 

58 seeded = 0 

59 already_set = 0 

60 no_region = 0 

61 no_bbox = 0 

62 for story in Story.objects.select_related('region').iterator(): 

63 if story.latitude is not None and story.longitude is not None: 

64 already_set += 1 

65 continue 

66 region = story.region 

67 if region is None: 

68 no_region += 1 

69 continue 

70 has_bbox = None not in ( 

71 region.bbox_north, region.bbox_south, 

72 region.bbox_east, region.bbox_west, 

73 ) 

74 if not has_bbox: 

75 no_bbox += 1 

76 continue 

77 rng = random.Random(story.id) 

78 lat, lng = _point_in_bbox(region, rng) 

79 story.latitude = lat 

80 story.longitude = lng 

81 story.save(update_fields=['latitude', 'longitude']) 

82 seeded += 1 

83 return seeded, already_set, no_region, no_bbox
